public class InstructiiDecizie {
    public static void main(String[] args) {
        int numar = 10;

        if (numar > 0) {
            System.out.println("Numarul este pozitiv.");
        } else if (numar < 0) {
            System.out.println("Numarul este negativ.");
        } else {
            System.out.println("Numarul este zero.");
        }
    }
}
